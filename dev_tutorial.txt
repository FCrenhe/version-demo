git checkout -b main # 切换分支
echo "Hello World - Stable version" > README.md
git add README.md

git commit -m "init: main stable version"

git push origin main


git checkout -b dev
echo "This is development version" > DEV.txt
git add DEV.txt
git commit -m "init: dev branch"

git push origin dev


git checkout -b test
echo "This is testing version" > TEST.txt
git add TEST.txt

git commit -m "init: test branch"
git push origin test



# 开发
git checkout dev
echo "New feature: voice recognition" >> DEV.txt
git add DEV.txt
git commit -m "feat: add voice recognition"
git push origin dev


#合并 test
git checkout test
git merge dev -m "merge: bring new feature from dev to test"
git push origin test

# 合并到主分支
git checkout main
git merge test -m "release: v1.1 add voice recognition"
git push origin main


# 打上版本标签

git tag -a v1.1 -m "Stable release v1.1"
git push origin v1.1
